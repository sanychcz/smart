def self.search(search)
  if search
    find(:all, :conditions => ['name LIKE ?', "%#{search}%"])
  else
    find(:all)
  end
end

def self.to_csv
  CSV.generate do |csv|
    csv << column_names
    all.each do |company|
      csv << company.attributes.values_at(*column_names)
    end
  end
end
